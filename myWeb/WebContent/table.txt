    "SIDO" VARCHAR2(10) NOT NULL,
    "GUGUN" VARCHAR2(30) NOT NULL,
    "DORO" VARCHAR2(50),

    "BDNAME" VARCHAR2(50),
    "DONG" VARCHAR2(50) NOT NULL,
